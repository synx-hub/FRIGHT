-- ===== GET PLAYER ROLE =====
local function GetPlayerRole(player)
    local char = player.Character
    if not char then return "Innocent" end
    
    local backpack = player.Backpack
    if backpack then
        for _, tool in ipairs(backpack:GetChildren()) do
            if tool:IsA("Tool") then
                if tool.Name == "Knife" then return "Murderer" end
                if tool.Name == "Gun" then return "Sheriff" end
            end
        end
    end
    
    for _, tool in ipairs(char:GetChildren()) do
        if tool:IsA("Tool") then
            if tool.Name == "Knife" then return "Murderer" end
            if tool.Name == "Gun" then return "Sheriff" end
        end
    end
    
    local success, module = pcall(function()
        local Modules = ReplicatedStorage:FindFirstChild("Modules")
        if Modules then
            local CurrentRound = Modules:FindFirstChild("CurrentRoundClient")
            if CurrentRound then
                local data = require(CurrentRound)
                if data and data.PlayerData then
                    local playerData = data.PlayerData[player.Name]
                    if playerData then
                        return playerData.Role or "Innocent"
                    end
                end
            end
        end
    end)
    
    return "Innocent"
end

local function IsAlive(player)
    local char = player.Character
    if not char then return false end
    local humanoid = char:FindFirstChildOfClass("Humanoid")
    return humanoid and humanoid.Health > 0
end

local function IsVisible(player)
    local char = player.Character
    if not char then return false end
    
    local head = char:FindFirstChild("Head")
    local root = char:FindFirstChild("HumanoidRootPart")
    if not head or not root then return false end
    
    local origin = Camera.CFrame.Position
    local target = Settings.Aimbot.Hitbox == "Head" and head.Position or root.Position
    
    local ray = Ray.new(origin, (target - origin) * 1.2)
    local hit = workspace:FindPartOnRayWithIgnoreList(ray, {LocalPlayer.Character, Camera})
    
    if not hit then return true end
    
    local charParts = {}
    for _, part in ipairs(char:GetDescendants()) do
        if part:IsA("BasePart") then
            charParts[part] = true
        end
    end
    
    return charParts[hit] ~= nil
end

local function GetPlayerWeapon(player)
    local char = player.Character
    if char then
        local tool = char:FindFirstChildOfClass("Tool")
        if tool then return tool.Name end
    end
    return "None"
end

-- ===== AIMBOT CORE =====
local function FindPlayersByRole(role)
    local found = {}
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and IsAlive(player) then
            if GetPlayerRole(player) == role then
                table.insert(found, player)
            end
        end
    end
    return found
end

local function GetBestTarget()
    local targetMode = Settings.Aimbot.TargetMode
    
    if AimbotState.LockedTarget then
        local player = AimbotState.LockedTarget
        if IsAlive(player) and IsVisible(player) then
            local role = GetPlayerRole(player)
            if targetMode == "Murderer" and role == "Murderer" then
                return player
            elseif targetMode == "Sheriff" and role == "Sheriff" then
                return player
            elseif targetMode == "Innocent" and role == "Innocent" then
                return player
            elseif targetMode == "Closest" then
                return player
            end
        end
        AimbotState.LockedTarget = nil
    end
    
    local bestTarget = nil
    local bestScore = -math.huge
    local mousePos = UserInputService:GetMouseLocation()
    local lockPriority = Settings.Aimbot.LockPriority or targetMode
    
    local priorityTargets = {}
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and IsAlive(player) then
            local role = GetPlayerRole(player)
            if role == lockPriority then
                table.insert(priorityTargets, player)
            end
        end
    end
    
    local targetsToCheck = priorityTargets
    if #priorityTargets == 0 then
        targetsToCheck = {}
        for _, player in ipairs(Players:GetPlayers()) do
            if player ~= LocalPlayer and IsAlive(player) then
                table.insert(targetsToCheck, player)
            end
        end
    end
    
    for _, player in ipairs(targetsToCheck) do
        local char = player.Character
        if not char then continue end
        
        local hitPart = Settings.Aimbot.Hitbox == "Head" and char:FindFirstChild("Head") or char:FindFirstChild("HumanoidRootPart")
        if not hitPart then continue end
        
        if Settings.Aimbot.WallCheck then
            local origin = Camera.CFrame.Position
            local target = hitPart.Position
            local ray = Ray.new(origin, (target - origin) * 1.2)
            local hit = workspace:FindPartOnRayWithIgnoreList(ray, {LocalPlayer.Character, Camera})
            if hit and not hitPart:IsDescendantOf(hit) then
                continue
            end
        end
        
        if Settings.Aimbot.VisibleCheck and not IsVisible(player) then
            continue
        end
        
        local role = GetPlayerRole(player)
        
        if targetMode ~= "Closest" then
            if targetMode == "Murderer" and role ~= "Murderer" then continue end
            if targetMode == "Sheriff" and role ~= "Sheriff" then continue end
            if targetMode == "Innocent" and role ~= "Innocent" then continue end
        end
        
        local screenPos, onScreen = Camera:WorldToViewportPoint(hitPart.Position)
        if not onScreen then continue end
        
        local screenVector = Vector2.new(screenPos.X, screenPos.Y)
        local distFromCenter = (screenVector - Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)).Magnitude
        
        if distFromCenter > Settings.Aimbot.FOV then continue end
        
        local score = 0
        
        if role == lockPriority then score = score + 1000 end
        if role == "Murderer" then score = score + 500
        elseif role == "Sheriff" then score = score + 250 end
        
        if targetMode == "Closest" then
            score = score - distFromCenter
        else
            score = score - distFromCenter
        end
        
        if AimbotState.TargetHistory[player] then
            score = score + 50
        end
        
        local distance = (LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") and 
                       (hitPart.Position - LocalPlayer.Character.HumanoidRootPart.Position).Magnitude or 999)
        score = score - (distance / 1000)
        
        if score > bestScore then
            bestScore = score
            bestTarget = player
        end
    end
    
    if bestTarget then
        for player, _ in pairs(AimbotState.TargetHistory) do
            AimbotState.TargetHistory[player] = (AimbotState.TargetHistory[player] or 0) - 1
            if AimbotState.TargetHistory[player] <= 0 then
                AimbotState.TargetHistory[player] = nil
            end
        end
        AimbotState.TargetHistory[bestTarget] = 10
        AimbotState.LockedTarget = bestTarget
    end
    
    return bestTarget
end

local function SmoothAim(target)
    local char = LocalPlayer.Character
    if not char then return end
    
    local hitPart = target.Character:FindFirstChild(Settings.Aimbot.Hitbox)
    if not hitPart then 
        hitPart = target.Character:FindFirstChild("HumanoidRootPart")
        if not hitPart then return end
    end
    
    local targetPos = hitPart.Position
    
    if Settings.Aimbot.Prediction then
        local velocity = target.Character:FindFirstChild("HumanoidRootPart") and 
                        target.Character.HumanoidRootPart.Velocity or Vector3.new(0, 0, 0)
        targetPos = targetPos + velocity * Settings.Aimbot.PredictionAmount
    end
    
    local camPos = Camera.CFrame.Position
    local lookAt = (targetPos - camPos).Unit
    
    local newCFrame = CFrame.new(camPos, camPos + lookAt)
    local smoothFactor = math.min(1, 1 / Settings.Aimbot.Smoothness * 2)
    Camera.CFrame = Camera.CFrame:Lerp(newCFrame, smoothFactor)
end

local function AddSlider(parent, text, min, max, default, settingTable, settingKey, callback)
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(1, 0, 0, 50)
    frame.BackgroundColor3 = Colors.Card
    frame.BorderSizePixel = 0
    frame.Parent = parent
    CreateCorner(frame, 8)
    CreateStroke(frame, Colors.Border, 1, 0)

    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(0.55, 0, 0, 20)
    label.Position = UDim2.new(0, 14, 0, 6)
    label.Text = text
    label.TextColor3 = Colors.Text
    label.Font = Enum.Font.GothamMedium
    label.TextSize = 12
    label.TextXAlignment = Enum.TextXAlignment.Left
    label.BackgroundTransparency = 1
    label.Parent = frame

    local valueLabel = Instance.new("TextLabel")
    valueLabel.Size = UDim2.new(0.35, -14, 0, 20)
    valueLabel.Position = UDim2.new(0.65, 0, 0, 6)
    valueLabel.Text = tostring(default)
    valueLabel.TextColor3 = Colors.AccentLight
    valueLabel.Font = Enum.Font.GothamBold
    valueLabel.TextSize = 12
    valueLabel.TextXAlignment = Enum.TextXAlignment.Right
    valueLabel.BackgroundTransparency = 1
    valueLabel.Parent = frame

    local track = Instance.new("TextButton")
    track.Size = UDim2.new(1, -28, 0, 5)
    track.Position = UDim2.new(0, 14, 1, -14)
    track.BackgroundColor3 = Colors.TrackBg
    track.Text = ""
    track.AutoButtonColor = false
    track.Parent = frame
    CreateCorner(track, 3)

    local fill = Instance.new("Frame")
    fill.Size = UDim2.new((default - min) / (max - min), 0, 1, 0)
    fill.BackgroundColor3 = Colors.Accent
    fill.BorderSizePixel = 0
    fill.Parent = track
    CreateCorner(fill, 3)

    local handle = Instance.new("Frame")
    handle.Size = UDim2.new(0, 14, 0, 14)
    handle.Position = UDim2.new((default - min) / (max - min), -7, 0.5, -7)
    handle.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    handle.BorderSizePixel = 0
    handle.Parent = track
    CreateCorner(handle, 7)
    CreateStroke(handle, Colors.Accent, 2, 0)

    local dragging = false

    local function UpdateSlider(input)
        local pos = math.clamp((input.Position.X - track.AbsolutePosition.X) / track.AbsoluteSize.X, 0, 1)
        local val = math.floor(min + (max - min) * pos)
        val = math.clamp(val, min, max)
        settingTable[settingKey] = val
        valueLabel.Text = tostring(val)
        fill.Size = UDim2.new(pos, 0, 1, 0)
        handle.Position = UDim2.new(pos, -7, 0.5, -7)
        if callback then callback(val) end
    end

    track.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            UpdateSlider(input)
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            UpdateSlider(input)
        end
    end)
end

        optBtn.Size = UDim2.new(1, -4, 0, 24)
        optBtn.BackgroundColor3 = option == selected and Colors.AccentDark or Colors.Card
        optBtn.Text = option

        optBtn.TextSize = 11

        CreateCorner(optBtn, 5)

        dropdown.Visible = not dropdown.Visible
    end)
end

RunService.RenderStepped:Connect(function()
    if Settings.Aimbot.ShowFOV and Drawing and not AimbotState._fovCircle then
        AimbotState._fovCircle = Drawing.new("Circle")
        AimbotState._fovCircle.Color = Colors.Accent
        AimbotState._fovCircle.Thickness = 1.5
        AimbotState._fovCircle.NumSides = 64
        AimbotState._fovCircle.Filled = false
        AimbotState._fovCircle.Transparency = 0.8
        AimbotState._fovCircle.Radius = Settings.Aimbot.FOV
        AimbotState._fovCircle.Visible = false
    end
    
    if AimbotState._fovCircle then
        local mouse = UserInputService:GetMouseLocation()
        AimbotState._fovCircle.Position = Vector2.new(mouse.X, mouse.Y)
        AimbotState._fovCircle.Radius = Settings.Aimbot.FOV
        AimbotState._fovCircle.Visible = Settings.Aimbot.Enabled and Settings.Aimbot.ShowFOV
    end
    
    if not Settings.Aimbot.Enabled then
        AimbotState.Target = nil
        AimbotState.Held = false
        if AimbotState._targetHighlight then
            AimbotState._targetHighlight.Enabled = false
        end
        return
    end
    
    local held = UserInputService:IsKeyDown(Settings.Aimbot.LockKey)
    if not held then
        AimbotState.Target = nil
        AimbotState.Held = false
        AimbotState.LockedTarget = nil
        if AimbotState._targetHighlight then
            AimbotState._targetHighlight.Enabled = false
        end
        return
    end
    
    local target = GetBestTarget()
    if not target then 
        AimbotState.Target = nil
        AimbotState.Held = false
        if AimbotState._targetHighlight then
            AimbotState._targetHighlight.Enabled = false
        end
        return 
    end
    
    SmoothAim(target)
    AimbotState.Target = target
    AimbotState.Held = true
    
    if target and target.Character then
        if not AimbotState._targetHighlight then
            AimbotState._targetHighlight = Instance.new("Highlight")
            AimbotState._targetHighlight.FillTransparency = 0.3
            AimbotState._targetHighlight.OutlineTransparency = 0.3
            AimbotState._targetHighlight.Parent = ScreenGui
        end
        
        local role = GetPlayerRole(target)
        if role == "Murderer" then
            AimbotState._targetHighlight.FillColor = Colors.RoleRed
            AimbotState._targetHighlight.OutlineColor = Colors.RoleRed
        elseif role == "Sheriff" then
            AimbotState._targetHighlight.FillColor = Colors.RoleBlue
            AimbotState._targetHighlight.OutlineColor = Colors.RoleBlue
        else
            AimbotState._targetHighlight.FillColor = Colors.RoleGreen
            AimbotState._targetHighlight.OutlineColor = Colors.RoleGreen
        end
        
        AimbotState._targetHighlight.Adornee = target.Character
        AimbotState._targetHighlight.Enabled = true
    end
